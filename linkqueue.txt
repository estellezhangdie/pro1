class Node():#定义结点
    def __init__(self,item=None,next=None):
        self.item=item
        self.next=next
class _LinkListQueue(object):
    def __init__(self):
        self.first=None
        self.last=None
        self.N=0
    def isEmpty(self):
        return self.first is None
    def size(self):
        return self.N
    def enqueue(self,item):
        node=Node(item)
        if self.isEmpty():
            self.first=node
        else:
            self.last.next=node
        self.last=node
        self.N+=1
    def dequeue(self):
        if self.first is None:
            print( "this queue is empty")
        item = self.first.item
        self.first=self.first.next
        self.N-=1
        return item
    def __iter__(self):
        return self.first
my_queue =_LinkListQueue()
my_queue.enqueue('h')
my_queue.enqueue('a')
